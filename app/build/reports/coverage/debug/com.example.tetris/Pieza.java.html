    private Pieza pieza;
    public int pos;

    public Pieza() {
    }

    public Pieza(Pieza pieza) {
        this.pieza = pieza;
        this.x1 = pieza.x1;
        this.x2 = pieza.x2;
        this.x3 = pieza.x3;
        this.x4 = pieza.x4;
        this.y1 = pieza.y1;
        this.y2 = pieza.y2;
        this.y3 = pieza.y3;
        this.y4 = pieza.y4;
    }

    public void mover(int x, int y) {
        x1 = x1 + x;
        y1 = y1 + y;
        x2 = x2 + x;
        y2 = y2 + y;
        x3 = x3 + x;
        y3 = y3 + y;
        x4 = x4 + x;
        y4 = y4 + y;
    }

    public int getAltura() {
        return this.y1;
    }

    public void setAltura(int y) {
        this.pieza.y1 += y;
        this.pieza.y2 += y;
        this.pieza.y3 += y;
        this.pieza.y4 += y;
    }

    public void copiarPieza(Pieza pieza2) {
        this.x1 = pieza2.x1;
        this.x2 = pieza2.x2;
        this.x3 = pieza2.x3;
        this.x4 = pieza2.x4;

        this.y1 = pieza2.y1;
        this.y2 = pieza2.y2;
        this.y3 = pieza2.y3;
        this.y4 = pieza2.y4;


        this.pos = pieza2.pos;

    }
